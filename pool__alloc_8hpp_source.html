270template <class T, std::size_t NBlock, template <typename> class BuffInitAlloc>
271 requires std::is_same_v<T, std::remove_reference_t<T>>
272class PoolAlloc<T, NBlock, BuffInitAlloc>::AllocBuf {
273public:
274 constexpr AllocBuf() noexcept {
275 AllocSlot* last_slot = nullptr;
276 // std::views::zip hasn't landed in libstdc++
277 // for (std::tuple<AllocSlot&, std::add_lvalue_reference_t<T>> slot_block
278 // :
279 // std::views::zip(m_slot_list, *m_buff) | std::views::reverse) {
280 // auto& slot = std::get<0>(slot_block);
281 // slot.blk = &std::get<1>(slot_block);
282 // // slot.blk = m_buff->data() + idx;
283 // slot.next = last_slot;
284 // last_slot = &slot;
285 // }
286
287 // MSVC doesn't allow direct conversion from iterator to pointer.
288 // In gcc/clang, an iterator seems to be just an alias to a pointer.
289 auto buff_iter =
290 // the minus behind m_buff.back() is to accomodate memory alignment.
291 reinterpret_cast<pointer>( // NOLINT(*reinterpret-cast*)
292 &m_buff.back() - (sizeof(T) - 1));
293 for (auto& slot : m_slot_list | std::views::reverse) {
294 slot.blk = buff_iter--;
295 slot.next = last_slot;
296 last_slot = &slot;
297 }
298 m_head.store(m_slot_list.data(), std::memory_order::release);
299 }
300 AllocBuf(const AllocBuf&) = delete;
301 AllocBuf(AllocBuf&&) = delete;
302 auto operator=(const AllocBuf&) = delete;
303 auto operator=(AllocBuf&&) = delete;
304 constexpr ~AllocBuf() = default;

319 std::array<uint8_t, sizeof(T) * NBlock> m_buff{};
326 std::array<AllocSlot, NBlock> m_slot_list{};
331 std::atomic<AllocSlot*> m_head{};
332};

410template <class T, std::size_t NBlock, template <typename> class BuffInitAlloc>
411 requires std::is_same_v<T, std::remove_reference_t<T>>
412[[nodiscard]] auto
413PoolAlloc<T, NBlock, BuffInitAlloc>::AllocBuf::allocate() noexcept -> pointer {
414 // wait until initialization finishes.
415 // Since @c notify_all on this flag is called with release,
416 // an acquire is to make sure that anything before the release
417 // is, in fact, done.
418
419 auto curr_head = m_head.load(std::memory_order::relaxed);
420 if (curr_head == nullptr) {
421 return nullptr;
422 }
423 // gain an exclusive block.
424 while (!m_head.compare_exchange_weak(curr_head, curr_head->next,
425 std::memory_order::acquire,
426 std::memory_order::relaxed)) {
427 // no more memory in this allocator.
428 if (curr_head == nullptr) {
429 return nullptr;
430 }
431 }
432 // now curr_head is an exclusive block.
433 return curr_head->blk;
434}

436template <class T, std::size_t NBlock, template <typename> class BuffInitAlloc>
437 requires std::is_same_v<T, std::remove_reference_t<T>>
438void PoolAlloc<T, NBlock, BuffInitAlloc>::AllocBuf::deallocate(
439 pointer t_p_obj) noexcept {
440 auto t_idx = static_cast<size_type>(
441 t_p_obj -
442 reinterpret_cast<pointer>(m_buff.data())); // NOLINT(*reinterpret-cast)
443 auto* slot = &m_slot_list.at(t_idx);
444 auto curr_head = m_head.load(std::memory_order::relaxed);
445 while (!m_head.compare_exchange_weak(curr_head, slot,
446 std::memory_order::release,
447 std::memory_order::relaxed)) {
448 }
449 slot->next.store(curr_head, std::memory_order::release);
450}

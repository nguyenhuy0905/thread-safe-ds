326template <class T, std::size_t NBlock, template <typename> class BuffInitAlloc>
327 requires std::is_same_v<T, std::remove_reference_t<T>>
328[[nodiscard]] auto
329PoolAlloc<T, NBlock, BuffInitAlloc>::AllocBuf::allocate() noexcept -> pointer {
330 // wait until initialization finishes.
331 // We don't need to get the "spinlock" here.
332 // Since notify_all on this flag is called with release,
333 // an acquire is to make sure that anything before the release
334 // is, in fact, done.
335 if (!m_init_flag.test(std::memory_order::acquire)) {
336 m_init_flag.wait(false, std::memory_order::relaxed);
337 }
338
339 auto curr_head = m_head.load(std::memory_order::relaxed);
340 if (curr_head == nullptr) {
341 return nullptr;
342 }
343 // gain an exclusive block.
344 while (!m_head.compare_exchange_weak(curr_head, curr_head->next,
345 std::memory_order::acquire,
346 std::memory_order::relaxed)) {
347 // no more memory in this allocator.
348 if (curr_head == nullptr) {
349 return nullptr;
350 }
351 }
352 // now curr_head is an exclusive block.
353 return curr_head->blk;
354}

356template <class T, std::size_t NBlock, template <typename> class BuffInitAlloc>
357 requires std::is_same_v<T, std::remove_reference_t<T>>
358void PoolAlloc<T, NBlock, BuffInitAlloc>::AllocBuf::deallocate(
359 pointer t_p_obj) noexcept {
360 auto t_idx = static_cast<size_type>(t_p_obj - m_buff->data());
361 auto* slot = &m_slot_list.at(t_idx);
362 auto curr_head = m_head.load(std::memory_order::relaxed);
363 while (!m_head.compare_exchange_weak(curr_head, slot,
364 std::memory_order::release,
365 std::memory_order::relaxed)) {
366 }
367 slot->next.store(curr_head, std::memory_order::release);
368}

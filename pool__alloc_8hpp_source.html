326 template <class T, std::size_t NBlock, template <typename> class BuffInitAlloc>
327  requires std::is_same_v<T, std::remove_reference_t<T>>
328 [[nodiscard]] auto
329 PoolAlloc<T, NBlock, BuffInitAlloc>::AllocBuf::allocate() noexcept -> pointer {
330  // wait until initialization finishes.
331  // We don't need to get the "spinlock" here.
332  if (!m_init_flag.test()) {
333  m_init_flag.wait(false, std::memory_order::relaxed);
334  }
335 
336  auto curr_head = m_head.load(std::memory_order::relaxed);
337  if (curr_head == nullptr) {
338  return nullptr;
339  }
340  // gain an exclusive block.
341  while (!m_head.compare_exchange_weak(curr_head, curr_head->next,
342  std::memory_order::acquire,
343  std::memory_order::relaxed)) {
344  // no more memory in this allocator.
345  if (curr_head == nullptr) {
346  return nullptr;
347  }
348  }
349  // now curr_head is an exclusive block.
350  return curr_head->blk;
351 }

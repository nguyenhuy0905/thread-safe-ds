241template <class T, std::size_t NBlock, template <typename> class BuffInitAlloc>
242 requires std::is_same_v<T, std::remove_reference_t<T>>
243class PoolAlloc<T, NBlock, BuffInitAlloc>::AllocBuf {
244public:
245 constexpr AllocBuf() noexcept {
246 AllocSlot* last_slot = nullptr;
247 // std::views::zip hasn't landed in libstdc++
248 // for (std::tuple<AllocSlot&, std::add_lvalue_reference_t<T>> slot_block
249 // :
250 // std::views::zip(m_slot_list, *m_buff) | std::views::reverse) {
251 // auto& slot = std::get<0>(slot_block);
252 // slot.blk = &std::get<1>(slot_block);
253 // // slot.blk = m_buff->data() + idx;
254 // slot.next = last_slot;
255 // last_slot = &slot;
256 // }
257
258 // MSVC doesn't allow direct conversion from iterator to pointer.
259 // In gcc/clang, an iterator seems to be just an alias to a pointer.
260 auto buff_iter =
261 // the minus behind m_buff.back() is to accomodate memory alignment.
262 reinterpret_cast<pointer>( // NOLINT(*reinterpret-cast*)
263 &m_buff.back() - (sizeof(T) - 1));
264 for (auto& slot : m_slot_list | std::views::reverse) {
265 slot.blk = buff_iter--;
266 slot.next = last_slot;
267 last_slot = &slot;
268 }
269 m_head.store(m_slot_list.data(), std::memory_order::release);
270 }
271 AllocBuf(const AllocBuf&) = delete;
272 AllocBuf(AllocBuf&&) = delete;
273 auto operator=(const AllocBuf&) = delete;
274 auto operator=(AllocBuf&&) = delete;
275 constexpr ~AllocBuf() = default;

290 std::array<uint8_t, sizeof(T) * NBlock> m_buff{};
297 std::array<AllocSlot, NBlock> m_slot_list{};
302 std::atomic<AllocSlot*> m_head{};
303};

398template <class T, std::size_t NBlock, template <typename> class BuffInitAlloc>
399 requires std::is_same_v<T, std::remove_reference_t<T>>
400[[nodiscard]] auto
401PoolAlloc<T, NBlock, BuffInitAlloc>::AllocBuf::allocate() noexcept -> pointer {
402 // wait until initialization finishes.
403 // Since @c notify_all on this flag is called with release,
404 // an acquire is to make sure that anything before the release
405 // is, in fact, done.
406
407 auto curr_head = m_head.load(std::memory_order::relaxed);
408 if (curr_head == nullptr) {
409 return nullptr;
410 }
411 // gain an exclusive block.
412 while (!m_head.compare_exchange_weak(curr_head, curr_head->next,
413 std::memory_order::acquire,
414 std::memory_order::relaxed)) {
415 // no more memory in this allocator.
416 if (curr_head == nullptr) {
417 return nullptr;
418 }
419 }
420 // now curr_head is an exclusive block.
421 return curr_head->blk;
422}

424template <class T, std::size_t NBlock, template <typename> class BuffInitAlloc>
425 requires std::is_same_v<T, std::remove_reference_t<T>>
426void PoolAlloc<T, NBlock, BuffInitAlloc>::AllocBuf::deallocate(
427 pointer t_p_obj) noexcept {
428 auto t_idx = static_cast<size_type>(
429 t_p_obj -
430 reinterpret_cast<pointer>(m_buff.data())); // NOLINT(*reinterpret-cast)
431 auto* slot = &m_slot_list.at(t_idx);
432 auto curr_head = m_head.load(std::memory_order::relaxed);
433 while (!m_head.compare_exchange_weak(curr_head, slot,
434 std::memory_order::release,
435 std::memory_order::relaxed)) {
436 }
437 slot->next.store(curr_head, std::memory_order::release);
438}

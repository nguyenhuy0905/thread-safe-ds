84 auto allocate(AllocInfo t_alloc_info) -> void* {
85 auto align_index = [&](std::size_t t_idx) -> std::size_t {
86 // similar to curr_head_num % curr_head_idx since we assume align is a
87 // power of 2
88 auto* head_ptr = &m_buff.front() + t_idx;
89 // NOLINTNEXTLINE(*reinterpret-cast*)
90 auto curr_head_num = reinterpret_cast<std::uintptr_t>(head_ptr);
91 auto mod = curr_head_num & (t_alloc_info.align - 1);
92 return (mod == 0)
93 ? t_idx
94 : static_cast<std::size_t>(t_idx + t_alloc_info.align - mod);
95 };
96
97 auto curr_head_idx = m_head_idx.load(std::memory_order::relaxed);
98 auto aligned_idx = align_index(curr_head_idx);
99 auto next_head_idx =
100 static_cast<std::size_t>(aligned_idx + t_alloc_info.size);
101 if (next_head_idx >= Size) {
102 return nullptr;
103 }
104
105 while (!m_head_idx.compare_exchange_weak(curr_head_idx, next_head_idx,
106 std::memory_order::release,
107 std::memory_order::acquire)) {
108 aligned_idx = align_index(curr_head_idx);
109 next_head_idx = static_cast<std::size_t>(aligned_idx + t_alloc_info.size);
110 if (next_head_idx >= Size) {
111 return nullptr;
112 }
113 }
114 return static_cast<void*>(&m_buff.front() + aligned_idx);
115 }
116
117private:

119 std::atomic<decltype(Size)> m_head_idx{};
